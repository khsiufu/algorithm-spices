class TreeNode(object):
    def __init__(self, x):
        self.val = x
        self.left = None
        self.right = None

def make_tree(lst):
    node = None
    if not lst:
        return
    if lst[0] is not None:
        node = TreeNode(lst.pop(0))
        node.left = make_tree(lst)
        node.right = make_tree(lst)
    else:
        lst.pop(0)
    return node

def inorder(root, res):
    if not root:
        return
    inorder(root.left, res)
    res.append(root.val)
    inorder(root.right, res)
